\begin{itemize}

    \item A buyer participates only if auction information reaches her.

\begin{frame}{Baseline properties we care about}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\greenblock{Incentive Compatibility (IC)}{
Truthful bidding and full diffusion is a dominant strategy:
\[
 u_i(v_i,r_i; a_{-i}) \ge u_i(v_i',r_i'; a_{-i}).
\]
}
\greenblock{Individual Rationality (IR)}{
A truthful participant gets non-negative utility.
}
\end{column}
\begin{column}{0.48\textwidth}
\greenblock{Weak Budget Balance / Non-deficit}{
Seller revenue is non-negative:
\[
\sum_i p_i(a) \ge 0.
\]
}
\greenblock{Social welfare}{
For one item, welfare is value of the winner. For $k$ items, sum of values of real winners.
}
\end{column}
\end{columns}

\begin{frame}{Two single-item mechanisms: VCG and IDM}

% add that in idm not necessary that highest bidder wins 


\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.49\textwidth}
\begin{block}{Network VCG}
\begin{itemize}
    \item Allocates to highest reachable bidder.
    \item Critical intermediates may receive rewards.
    \item Efficient and IC.
    \item But seller can run a deficit.
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.49\textwidth}
\begin{block}{IDM}
\begin{itemize}
    \item Uses diffusion critical sequence.
    \item Rewards only price-improving diffusion.
    \item IC, IR, non-deficit.
    \item May sacrifice welfare compared with VCG.
\end{itemize}
\end{block}
\end{column}
\end{columns}

\begin{column}{0.48\textwidth}
\begin{itemize}
    \item A real buyer creates fake identities.
    \item Fakes can connect internally or to the creator's neighbors.
    \item No collusion: fakes are controlled by one real buyer.
    \item Goal: increase allocation chance or payment/reward.
\end{itemize}
\greenblock{In diffusion auctions}{Sybil nodes can manipulate the very paths used to define criticality and rewards.}
\end{column}
\end{columns}
{\tiny Sybil model follows Chen, Deng, Wang, Wu, Zhao, \emph{Sybil-Proof Diffusion Auction in Social Networks}, AAMAS 2023.}
\end{frame}

\begin{frame}{Definition: Sybil-proofness}
\begin{definition}[Sybil-proofness, informal]
A mechanism is \impterm{Sybil-proof} if a buyer cannot increase her utility by replacing herself with a set of fake identities.
\end{definition}
\vspace{2mm}
\greenblock{Formal shape}{For every real buyer $x$, every profile of others $\theta_{-x}$, and every Sybil attack $\sigma_x$,
\[
    u_x(\text{truthful }x,\theta_{-x})
    \ge
    u_x(\text{identities created by }\sigma_x,\theta_{-x}).
\]
}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.48\textwidth}
\begin{itemize}
    \item Utility aggregates all identities controlled by $x$.
    \item Fakes may bid and diffuse strategically.
\end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
\begin{itemize}
    \item No collusion between different real buyers.
    \item This is still a strong requirement.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{frame}{Can single-item diffusion auctions be made Sybil-proof?}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.49\textwidth}
\begin{block}{STM: Sybil Tax Mechanism}
\begin{itemize}
    \item Identifies trusted/non-Sybil roots.
    \item Makes suspicious diffusion less profitable.
    \item Strong revenue behavior.
    \item Weaker diffusion incentive to suspicious parts.
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.49\textwidth}
\begin{block}{SCM: Sybil Cluster Mechanism}
\begin{itemize}
    \item Selects/uses a graph structure to limit fake influence.
    \item Credits reachability to selected real agents.
    \item Better diffusion motivation.
    \item Mild welfare/revenue sacrifice.
\end{itemize}
\end{block}
\end{column}
\end{columns}

{\tiny Chen et al. propose STM and SCM for the single-item setting and explicitly leave multi-item Sybil-proof diffusion auctions as an open problem.}
\end{frame}

% -----------------------------------------------------------------------------
\begin{frame}{STM algorithm (single-item): pseudocode}
\scriptsize
\begin{enumerate}
  \item Given reported type profile $\theta'$, compute reachable reported graph $G_s(\theta')$ and graphical non-Sybil set $\Gamma(\theta')$. Write them as $G_s$ and $\Gamma$.
  \item Find highest-bid reachable buyer
  \[
    x^* \in \arg\max_{x \in V(G_s)} v_x'.
  \]
  \item Compute dominator sequence
  \[
    C_{x^*}=\{c_0=s,c_1,\ldots,c_\ell=x^*\},\qquad c_j=\mathrm{idom}(c_{j+1}),\;0\le j<\ell.
  \]
  \item Define buying and selling prices:
  \[
    p_j = \max\!\left[V(G_s)\setminus \alpha(c_j)\right],\quad 1<j\le \ell,
  \]
  \[
    q_j = \max\!\left[(V(G_s)\setminus \alpha(c_j))\cup \beta_j\right],\quad 1\le j<\ell,
  \]
  where
  \[
    \beta_j = \{x\mid \exists y\ne c_j,\; y\in (\alpha(c_j)\setminus\alpha(c_{j+1}))\cap\Gamma,\; x\in\alpha(y)\}.
  \]
  \item Pick the smallest index $d$ such that $v'_{c_d}\ge q_d$ for $1\le d<\ell$. If none exists, set $d=\ell$. Allocate item to $c_d$:
  \[
    \pi_{c_d}(\theta')=1.
  \]
  \item Payments:
  \[
    t_{c_j}(\theta')=
    \begin{cases}
      p_j-q_j, & 1\le j<d,\\
      p_j, & j=d.
    \end{cases}
  \]
  All other buyers have zero allocation and zero payment.
\end{enumerate}
\end{frame}

% -----------------------------------------------------------------------------
\begin{frame}{SCM algorithm (single-item): pseudocode}
\scriptsize
\begin{enumerate}
  \item Given reported type profile $\theta'$ as input, reconstruct a social graph on non-Sybil agents:
  \[
    H=(\Gamma(\theta'),E(H)),
  \]
  \[
    E(H)=\{(x,y)\mid \exists i\in K_x,\; \exists j\in K_y\;\text{s.t.}\; (i,j)\in E(G_s(\theta'))\}.
  \]
  \item Sample a random shortest-path tree of $H$ with equal probability, denoted by $T_H$.
  \item Construct a subgraph $\hat G$ of $G_s(\theta')$ using $T_H$:
  \[
    \hat G=(V(G_s(\theta')),E(\hat G)),
  \]
  \[
    E(\hat G)=E(G_s(\theta'))\setminus\{(i,j)\mid i\in K_x,\; j\in K_y,\; (x,y)\in E(H)\setminus E(T_H)\}.
  \]
  Equivalently, delete edge $(i,j)$ if $i\in K_x$, $j\in K_y$, $x\ne y$, and $(x,y)\notin T_H$.
  \item Run STM on the modified inputs
  \[
    G_s=\hat G,\qquad \Gamma=\Gamma(\theta')
  \]
  instead of
  \[
    G_s=G_s(\theta'),\qquad \Gamma=\Gamma(\theta').
  \]
\end{enumerate}
\end{frame}

% -----------------------------------------------------------------------------
\begin{frame}{STM vs SCM: useful intuition for our extension}
\begin{alertblock}{Lemma-level takeaway from STM}
When $\Gamma_0=\emptyset$, every buyer except the winner has payment $0$, hence utility $0$ (Lemma 4.2).\
In short: \impterm{intermediaries/brokers do not benefit from telling auction information in STM}.
\end{alertblock}

\vspace{1mm}
\begin{enumerate}
  \item \textbf{STM issue (shared trust, no attribution):} non-Sybil discoveries are pooled in $\Gamma$, so for many brokers the benchmark prices coincide ($p_j=q_j$). Therefore broker reward is $q_j-p_j=0$.

  \item \textbf{SCM fix (credit attribution by pruning):} SCM builds $H$, samples a shortest-path tree $T_H$, and removes redundant inter-cluster edges. This makes specific brokers the exclusive gateway for some trustworthy clusters.

  \item \textbf{Payment impact:} after pruning, those trustworthy buyers contribute to $q_j$ but are not available in $p_j$, creating a positive gap ($q_j>p_j$). Brokers can now get positive utility, so diffusion incentives are stronger than in STM.
\end{enumerate}

\greenblock{Interpretation}{STM mainly taxes suspicious influence; SCM preserves Sybil-proofness while restoring broker incentives via structure-aware edge removal.}
\end{frame}

\begin{frame}{SC-GIDM formalization: Phase 1 and Phase 2}
\small
\begin{block}{Phase 1: structural abstraction via dominator clusters}
Given reported graph $G(\theta')$, compute dominator tree $T_{dom}$ rooted at seller $s$.
For each node $i$ with $\mathrm{idom}(i)=s$, define
\[
  V_i = \{j\in N\mid i\text{ dominates }j\}\cup\{i\}.
\]
This yields disjoint clusters $\Ccal=\{V_1,\dots,V_m\}$ and a macro DAG $G_{cluster}$ with
\[
  (V_i,V_j)\in E(G_{cluster})\iff \exists u\in V_i,\exists v\in V_j:(u,v)\in E(G(\theta')).
\]
\end{block}

\begin{block}{Phase 2: capacity restriction and bid aggregation}
For each cluster $V_i$, enforce unit capacity and define effective bid
\[
  v^*_{V_i}=\max_{j\in V_i} v'_j, \qquad \text{cap}(V_i)=1.
\]
Hence internal Sybil volume does not change macro demand: many fake nodes still induce one macro bidder.
\end{block}

\greenblock{Topological insight}{Under non-colluding Sybil attacks, fake identities remain inside the attacker's cluster; they do not create new external incoming edges into that cluster.}
\end{frame}

% -----------------------------------------------------------------------------
\begin{frame}{SC-GIDM formalization: Phase 3 and Phase 4}
\small
\begin{block}{Phase 3: clustered GIDM allocation on $G_{cluster}$}
Run GIDM on macro nodes $(V_i,v^*_{V_i})$ with $K$ items:
\begin{itemize}
  \item compute top-$K$ macro winners and the critical macro allocation structure,
  \item route items by GIDM's DFS/stack logic,
  \item apply macro displacement when an upstream macro node outbids downstream critical nodes.
\end{itemize}
\end{block}

\begin{block}{Phase 4: settlement and intra-cluster assignment}
If cluster $V_i$ wins, assign the physical item to a node $j\in V_i$ with $v'_j=v^*_{V_i}$.
Cluster-level transfer (GIDM form on macro graph):
\[
  x_{V_i}=SW^{macro}_{-D_{V_i}}-\big(SW^{macro}_{-C^K_{V_i}}-v^*_{V_i}\big)
\]
for winning clusters, and
\[
  x_{V_i}=SW^{macro}_{-D_{V_i}}-SW^{macro}_{-C^K_{V_i}}
\]
for intermediary clusters. (Sign convention follows GIDM transfer notation.)
\end{block}
\end{frame}

\begin{frame}{Assumptions: what exactly are we proving?}
\begin{columns}[T,totalwidth=\textwidth]
\begin{column}{0.49\textwidth}
\begin{block}{Auction side}
\begin{itemize}
    \item $k$ homogeneous items.
    \item Single-unit demand.
    \item Quasi-linear utilities.
    \item No collusion between real buyers.
\end{itemize}
\end{block}
\end{column}
\begin{column}{0.49\textwidth}
\begin{block}{Sybil side}
\begin{itemize}
    \item Fake identities are controlled by one buyer.
    \item Fakes have no incoming edges except through creator/fakes.
    \item Perfect cluster oracle for theorem.
    \item Noisy clusters only in simulations.
\end{itemize}
\end{block}
\end{column}
\end{columns}
\vspace{2mm}
\begin{alertblock}{Important honesty point}
We are not solving the general Sybil detection problem. We are proving that \emph{if} a trusted-root / cluster layer is available, then GIDM can be lifted safely in a restricted multi-item model.
\end{alertblock}
\end{frame}

% -----------------------------------------------------------------------------
\begin{frame}{Correctness verification for SC-GIDM}
\begin{theorem}[Reduction under assumptions]
With perfect cluster recovery, no collusion, and unit-demand physical agents, SC-GIDM on the reported graph is strategically equivalent to running GIDM on the induced cluster graph.
\end{theorem}

\vspace{1mm}
\begin{enumerate}
  \item \textbf{IR:} cluster-level winner utility is nonnegative by inherited GIDM threshold logic; thus truthful physical agents are ex-post IR.
  \item \textbf{DSIC (cluster level):} once clustering is fixed, each cluster behaves as one bidder in GIDM; truthful effective reporting is dominant.
  \item \textbf{Sybil-proofness (non-colluding split attacks):} fake identities remain internal to one cluster, capacity stays $1$, and macro routing/pricing depends only on cluster-level structure and $v^*_{V_i}$.
\end{enumerate}

\greenblock{Important caveat}{If clustering is noisy or collusion creates cross-cluster coordination, guarantees become approximate/empirical rather than theorem-level.}
\end{frame}

\begin{frame}{What we expect to learn}
\begin{itemize}
    \item In the no-attack setting, GIDM should usually be better than local sale because diffusion reaches more buyers.
    \item Under Sybil attacks, ordinary GIDM can be manipulated because it runs on reported identities.
    \item SC-GIDM should reduce attacker gain under good clustering.
    \item False negatives hurt robustness; false positives hurt welfare.
    \item Dense graphs may lose less welfare because there are alternate diffusion paths.
\end{itemize}
